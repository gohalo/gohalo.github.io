默认配置文件为 /etc/cargo/bastion/bastion.ini，可以在启动时通过 --config=xxx 指定新的路径。

# 日志级别，支持 debug info warn 等
LogLevel = info
# 日志保存路径
LogPath = /var/log/cargo/bastion/bastion.log
# 审计日志保存路径
RecordBasePath = /var/log/cargo/bastion/record

# SSH服务监听端口
SSHListener = :9000
# HTTP服务监听端口
HTTPListener = :9080
# 数据库连接信息
DatabaseURI = user:password@tcp(127.0.0.1:8027)/bastion?charset=utf8&parseTime=True

[http]
# 是否开启
Enabled = true
# 是否打印调试信息，包括头、请求、返回等
DumpBody = false
DumpBodyWithHeader = X-Dump-Body

# HTTP相关审计日志
AuditPath = /var/log/cargo/bastion/audit/http.log
AuditRotate = 50,4,3,true
AuditContent =

CORSAllowHeaders = *

# CORS允许的域名信息，通过 , 分割
CORSAllowOrigins = *
# CORS 允许的方法
CORSAllowMethods = GET,HEAD,PUT,PATCH,POST,DELETE
# CORS 允许配置的 Access-Control-Expose-Headers 参数
CORSExposeHeaders = X-Token-Remain
# CORS 对应的 MaxAge 参数，单位是秒
CORSMaxAge = 0

# JWT签名算法，支持 EdDSA HS256/384/512 PS256/384/512 ES256/384/512 RS256/384/512
# 详见 JWT 代码中的 RegisterSigningMethod() 实现
JWTSigningMethod = HS256
# JWT过期时间，支持 h m s 对应时分秒，最小为30m
JWTExpiry = 2h
# JWT加密用密钥，分别为明文以及加密配置
JWTSigningKey = xxxx
JWTSigningKeySecret = xxx
# JWT过期阈值，超过该值之后会设置 X-Token-Expiry 值，最小1m
JWTExpiryRemain = 15m
# JWT最大值，从登录开始，允许刷新
JWTMaxAge = 48h
